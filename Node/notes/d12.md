# Day 12: HTTP Server and Basic Express

## HTTP Protocol Basics

**HTTP** (HyperText Transfer Protocol) is the protocol used for communication between clients (browsers) and servers. It's a **request-response protocol**: the client sends a request, and the server sends back a response.

### HTTP Request Structure

```plain
GET /api/users HTTP/1.1
Host: example.com
Content-Type: application/json
Authorization: Bearer token123

{
  "name": "John"
}
```

**Components:**

1. **Method:** GET, POST, PUT, DELETE, PATCH, etc.
2. **URL Path:** `/api/users`
3. **HTTP Version:** HTTP/1.1 or HTTP/2
4. **Headers:** Key-value pairs with metadata (Content-Type, Authorization, etc.)
5. **Body:** Optional data (usually JSON)

### HTTP Response Structure

```plain
HTTP/1.1 200 OK
Content-Type: application/json
Content-Length: 50

{
  "id": 1,
  "name": "John",
  "email": "john@example.com"
}
```

**Components:**

1. **Status Code:** 200 (OK), 404 (Not Found), 500 (Server Error), etc.
2. **Status Message:** Human-readable description
3. **Headers:** Metadata about the response
4. **Body:** The actual response data

### Common HTTP Status Codes

```plain
2xx - Success
  200 OK - Request succeeded
  201 Created - Resource created successfully
  204 No Content - Success but no content to return

3xx - Redirection
  301 Moved Permanently
  302 Found (temporary redirect)
  304 Not Modified

4xx - Client Error
  400 Bad Request - Invalid syntax
  401 Unauthorized - Authentication required
  403 Forbidden - Authenticated but not allowed
  404 Not Found - Resource doesn't exist

5xx - Server Error
  500 Internal Server Error
  502 Bad Gateway
  503 Service Unavailable
```

## Creating a Basic HTTP Server with Node.js

### Native HTTP Module

```typescript
import http from "http";

const server = http.createServer((req, res) => {
  console.log(`${req.method} ${req.url}`);

  // Set response headers
  res.setHeader("Content-Type", "application/json");

  // Send response
  if (req.url === "/") {
    res.statusCode = 200;
    res.end(JSON.stringify({ message: "Welcome!" }));
  } else if (req.url === "/api/users") {
    res.statusCode = 200;
    res.end(
      JSON.stringify([
        { id: 1, name: "Alice" },
        { id: 2, name: "Bob" },
      ])
    );
  } else {
    res.statusCode = 404;
    res.end(JSON.stringify({ error: "Not Found" }));
  }
});

const PORT = 3000;
server.listen(PORT, () => {
  console.log(`Server running at http://localhost:${PORT}`);
});
```

### Problems with Native HTTP Module

- **No routing:** You manually check `req.url` with if/else
- **No middleware:** Can't easily add cross-cutting concerns (logging, auth, etc.)
- **No body parsing:** Manually parsing JSON is tedious
- **Boilerplate heavy:** Lots of repetitive code

## Express: Lightweight Web Framework

**Express** is a minimal and flexible Node.js web framework that makes building HTTP servers much easier.

### Basic Express Server

```typescript
import express, { Express, Request, Response } from "express";

const app: Express = express();
const PORT = 3000;

// Middleware: Parse JSON request bodies
app.use(express.json());

// Route: GET /
app.get("/", (req: Request, res: Response) => {
  res.json({ message: "Welcome to Express!" });
});

// Route: GET /api/users
app.get("/api/users", (req: Request, res: Response) => {
  res.json([
    { id: 1, name: "Alice" },
    { id: 2, name: "Bob" },
  ]);
});

// Route: GET /api/users/:id (Dynamic parameters)
app.get("/api/users/:id", (req: Request, res: Response) => {
  const userId = req.params.id;
  res.json({ id: userId, name: "User Details" });
});

// Route: POST /api/users (Create new user)
app.post("/api/users", (req: Request, res: Response) => {
  const newUser = req.body;
  res.status(201).json({ message: "User created", user: newUser });
});

// 404 Handler (Must be last)
app.use((req: Request, res: Response) => {
  res.status(404).json({ error: "Route not found" });
});

// Start server
app.listen(PORT, () => {
  console.log(`Server running at http://localhost:${PORT}`);
});
```

### Key Express Concepts

#### 1. Middleware

Middleware is a function that has access to the request and response objects and can perform tasks before sending the response.

```typescript
// Logger middleware
app.use((req: Request, res: Response, next) => {
  const start = Date.now();

  res.on("finish", () => {
    const duration = Date.now() - start;
    console.log(`${req.method} ${req.url} - ${duration}ms`);
  });

  next(); // Pass control to next middleware/route
});

// Authentication middleware
app.use((req: Request, res: Response, next) => {
  const token = req.headers.authorization;

  if (!token) {
    return res.status(401).json({ error: "Unauthorized" });
  }

  next();
});

// Apply middleware to specific route
app.get("/protected", authenticate, (req: Request, res: Response) => {
  res.json({ message: "This is protected" });
});
```

#### 2. Route Parameters and Query Strings

```typescript
// Route parameters: /users/123
app.get("/users/:id", (req: Request, res: Response) => {
  const userId = req.params.id;
  res.json({ userId });
});

// Query strings: /search?name=John&age=30
app.get("/search", (req: Request, res: Response) => {
  const { name, age } = req.query;
  res.json({ search: { name, age } });
});
```

#### 3. Request Body

```typescript
// POST /users with JSON body
app.post("/users", (req: Request, res: Response) => {
  const { name, email } = req.body;

  if (!name || !email) {
    return res.status(400).json({ error: "Missing fields" });
  }

  res.status(201).json({ message: "User created", user: { name, email } });
});
```

#### 4. Response Methods

```typescript
// JSON response
res.json({ key: "value" });

// Status code and JSON
res.status(201).json({ message: "Created" });

// Plain text
res.send("Hello World");

// Redirect
res.redirect("/home");

// Send file
res.download("file.pdf");

// Custom headers
res.set("Custom-Header", "value");
```

### Building a Simple API with Express

```typescript
import express, { Express, Request, Response } from "express";

interface Task {
  id: number;
  title: string;
  completed: boolean;
}

const app: Express = express();
app.use(express.json());

let tasks: Task[] = [
  { id: 1, title: "Learn Express", completed: false },
  { id: 2, title: "Build API", completed: true },
];

let nextId = 3;

// GET all tasks
app.get("/tasks", (req: Request, res: Response) => {
  res.json(tasks);
});

// GET task by ID
app.get("/tasks/:id", (req: Request, res: Response) => {
  const task = tasks.find((t) => t.id === parseInt(req.params.id));

  if (!task) {
    return res.status(404).json({ error: "Task not found" });
  }

  res.json(task);
});

// POST create task
app.post("/tasks", (req: Request, res: Response) => {
  const { title } = req.body;

  if (!title) {
    return res.status(400).json({ error: "Title is required" });
  }

  const newTask: Task = {
    id: nextId++,
    title,
    completed: false,
  };

  tasks.push(newTask);
  res.status(201).json(newTask);
});

// PUT update task
app.put("/tasks/:id", (req: Request, res: Response) => {
  const task = tasks.find((t) => t.id === parseInt(req.params.id));

  if (!task) {
    return res.status(404).json({ error: "Task not found" });
  }

  if (req.body.title) task.title = req.body.title;
  if (req.body.completed !== undefined) task.completed = req.body.completed;

  res.json(task);
});

// DELETE task
app.delete("/tasks/:id", (req: Request, res: Response) => {
  const index = tasks.findIndex((t) => t.id === parseInt(req.params.id));

  if (index === -1) {
    return res.status(404).json({ error: "Task not found" });
  }

  const deletedTask = tasks.splice(index, 1)[0];
  res.json({ message: "Task deleted", task: deletedTask });
});

app.listen(3000, () => {
  console.log("Server running at http://localhost:3000");
});
```

## Express Application Structure

For larger applications, organize your code:

```plain
src/
├── routes/
│   ├── tasks.ts
│   └── users.ts
├── middleware/
│   ├── auth.ts
│   └── errorHandler.ts
├── controllers/
│   ├── taskController.ts
│   └── userController.ts
├── models/
│   ├── Task.ts
│   └── User.ts
├── app.ts
└── index.ts
```

### Example: routes/tasks.ts

```typescript
import { Router, Request, Response } from "express";

const router = Router();

// GET /tasks
router.get("/", (req: Request, res: Response) => {
  res.json({ message: "Get all tasks" });
});

// POST /tasks
router.post("/", (req: Request, res: Response) => {
  res.status(201).json({ message: "Create task" });
});

export default router;
```

### Example: app.ts

```typescript
import express from "express";
import tasksRouter from "./routes/tasks";

const app = express();

app.use(express.json());
app.use("/tasks", tasksRouter);

export default app;
```

### Example: index.ts

```typescript
import app from "./app";

const PORT = 3000;

app.listen(PORT, () => {
  console.log(`Server running at http://localhost:${PORT}`);
});
```

## Summary

- **HTTP** is the protocol for web communication with requests and responses.
- **Node's native `http` module** works but requires lots of manual handling.
- **Express** provides a framework that simplifies routing, middleware, and response handling.
- **Middleware** are functions that process requests before they reach route handlers.
- **Routing** is clean and intuitive with Express.
- **Separation of concerns** (routes, controllers, models) makes code maintainable as it grows.
